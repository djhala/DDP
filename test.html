Analysis of economic and human impacts of severe weather

Synopsis

Data from the NOAA is analyzed in order to determine which events had the greatest impact in terms of health and economics. The greatest loss of human life is determined to come from from Tornados. The greatest economic damages are determined to result from Flooding.

Data Processing

Data is first downloaded from “https://d396qusza40orc.cloudfront.net/repdata/data/StormData.csv.bz2”. The data is then read in and “summary” data is removed.

stormdat <- read.csv("StormData.csv.bz2")
stormdat$EVTYPE <- as.character(stormdat$EVTYPE)
stormdat <- stormdat[!grepl("summary", stormdat$EVTYPE, ignore.case = T), ]

Following this, we group the weather events into 16 categories. This task is complicated by inconsistent event names and typos in the data. About 40 events do not fit conveniently into any of the categories, however they do not have large values for the statistics we're interested in evaluating.

stormdat[grepl("torn|whirlwind|nado|funnel|spout", stormdat$EVTYPE, ignore.case = T), 
    ]$EVTYPE <- "Tornado"
stormdat[grepl("floo|fld|high water|dam|rising water", stormdat$EVTYPE, ignore.case = T), 
    ]$EVTYPE <- "Flooding"
stormdat[grepl("fire|smoke", stormdat$EVTYPE, ignore.case = T), ]$EVTYPE <- "Fire"
stormdat[grepl("dry|drie|drou", stormdat$EVTYPE, ignore.case = T), ]$EVTYPE <- "Drought"
stormdat[grepl("tide|current|surf|wave|sea|tsu|swel|surge", stormdat$EVTYPE, 
    ignore.case = T), ]$EVTYPE <- "Ocean"
stormdat[grepl("heat|cold|warm|cool|hot|temp|hyp|record", stormdat$EVTYPE, ignore.case = T), 
    ]$EVTYPE <- "Temperature"
stormdat[grepl("rain|wet|Preci|slee|show", stormdat$EVTYPE, ignore.case = T), 
    ]$EVTYPE <- "Rain"
stormdat[grepl("vol", stormdat$EVTYPE, ignore.case = T), ]$EVTYPE <- "Volcano"
stormdat[grepl("hurr|depress|floyd|tropic|typh", stormdat$EVTYPE, ignore.case = T), 
    ]$EVTYPE <- "Cyclones"
stormdat[grepl("ava|slide", stormdat$EVTYPE, ignore.case = T), ]$EVTYPE <- "Slides"
stormdat[grepl("thun|light|tstm|lign", stormdat$EVTYPE, ignore.case = T), ]$EVTYPE <- "Lightning"
stormdat[grepl("hail", stormdat$EVTYPE, ignore.case = T), ]$EVTYPE <- "Hail"
stormdat[grepl("snow|ice|bliz|wint|fro|free|icy", stormdat$EVTYPE, ignore.case = T), 
    ]$EVTYPE <- "Winter Weather"
stormdat[grepl("fog|vog", stormdat$EVTYPE, ignore.case = T), ]$EVTYPE <- "Fog"
stormdat[grepl("dust", stormdat$EVTYPE, ignore.case = T), ]$EVTYPE <- "Dust"
stormdat[grepl("wind|wnd", stormdat$EVTYPE, ignore.case = T), ]$EVTYPE <- "Wind"

Next we adjust the values for property and crop damages to their actual values. This is done by multiplying the values in PROPDMG and CROPDMG by a factor of 10x according to value in the corresponding PROPDMGEXP or CROPDMGEXP column.

stormdat$PROPDMGEXP <- tolower(stormdat$PROPDMGEXP)
stormdat[stormdat$PROPDMGEXP == "h", ]$PROPDMG <- stormdat[stormdat$PROPDMGEXP == 
    "h", ]$PROPDMG * 100
stormdat[stormdat$PROPDMGEXP == "k", ]$PROPDMG <- stormdat[stormdat$PROPDMGEXP == 
    "k", ]$PROPDMG * 1000
stormdat[stormdat$PROPDMGEXP == "m", ]$PROPDMG <- stormdat[stormdat$PROPDMGEXP == 
    "m", ]$PROPDMG * 1e+06
stormdat[stormdat$PROPDMGEXP == "b", ]$PROPDMG <- stormdat[stormdat$PROPDMGEXP == 
    "b", ]$PROPDMG * 1e+09
digits = grepl("[[:digit:]]", stormdat$PROPDMGEXP)
stormdat[digits, ]$PROPDMG <- stormdat[digits, ]$PROPDMG * 10^as.numeric(stormdat[digits, 
    ]$PROPDMGEXP)
stormdat$CROPDMGEXP <- tolower(stormdat$CROPDMGEXP)
stormdat[stormdat$CROPDMGEXP == "h", ]$CROPDMG <- stormdat[stormdat$CROPDMGEXP == 
    "h", ]$CROPDMG * 100
stormdat[stormdat$CROPDMGEXP == "k", ]$CROPDMG <- stormdat[stormdat$CROPDMGEXP == 
    "k", ]$CROPDMG * 1000
stormdat[stormdat$CROPDMGEXP == "m", ]$CROPDMG <- stormdat[stormdat$CROPDMGEXP == 
    "m", ]$CROPDMG * 1e+06
stormdat[stormdat$CROPDMGEXP == "b", ]$CROPDMG <- stormdat[stormdat$CROPDMGEXP == 
    "b", ]$CROPDMG * 1e+09
digits = grepl("[[:digit:]]", stormdat$CROPDMGEXP)
stormdat[digits, ]$CROPDMG <- stormdat[digits, ]$CROPDMG * 10^as.numeric(stormdat[digits, 
    ]$CROPDMGEXP)

Next we use a for loop to create a new data frame, which will consist of summary data for each event type. The summary data will include the number of injuries, the number of fatalities, and the total dollar value of damages to crops and property. We then create three new small data frames with the top 8 event types for each of the data summarized.

mdat <- matrix(ncol = 4)
mdat <- as.data.frame(mdat)
colnames(mdat) <- c("Injuries", "Fatalities", "Total_Damage", "Event_Type")
for (i in unique(stormdat$EVTYPE)) {
    k <- subset(stormdat, stormdat$EVTYPE == i)
    ndat <- c(sum(k$INJURIES), sum(k$FATALITIES), sum(k$PROPDMG) + sum(k$CROPDMG), 
        i)
    mdat <- rbind(mdat, ndat)
}
mdat[[1]] <- as.numeric(mdat[[1]])
mdat[[2]] <- as.numeric(mdat[[2]])
mdat[[3]] <- as.numeric(mdat[[3]])
mdat <- subset(mdat, mdat$Event_Type != 0)
Inj <- tail(mdat[order(mdat$Injuries), ], 8)
Fat <- tail(mdat[order(mdat$Fatalities), ], 8)
Dam <- tail(mdat[order(mdat$Total_Damage), ], 8)
